
= Graphics
:authors: Cloudberry Pi Foundations
:toc:

Generate graphs using matplotlib only by entering simple commands.

== Example Usage

[source,python]
----
from graphs import Graphs

g = Graphs(
    title="Animal Balance",
    theme="ggplot"
)

g.pie_chart(
    labels=["Humans", "Animals", "Insects"],
    values=[80, 50, 40]
)

g.bar_graph(
    labels=["Humans", "Animals", "Insects"],
    values=[80, 50, 40]
)

g.horizontal_bar_graph(
    labels=["Humans", "Animals", "Insects"],
    values=[80, 50, 40]
)
----

== Creating a Graph Object

[source,python]
----
from graphs import Graphs

g = Graphs(
    title="My Graph",
    figsize=(8, 5),
    theme="ggplot",
    grid=True
)
----

=== Parameters

* title → Graph title
* figsize → Size of figure
* theme → Matplotlib style
* grid → Show grid

=== 3D Surface Graph

[source,python]
----

x = np.linspace(-5,5,50)
y = np.linspace(-5,5,50)
X, Y = np.meshgrid(x,y)
Z = np.sin(np.sqrt(X**2 + Y**2))

g.surface_3d_graph(X, Y, Z)

== Full Example

[source,python]
----
from graphs import Graphs

g = Graphs(
    title="Sales Data",
    theme="ggplot"
)

g.bar_graph(
    labels=["India","USA","China"],
    values=[100,200,300]
)

g.save("sales.png")
